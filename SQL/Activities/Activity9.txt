SELECT customer_id,
       customer_name,
       grade,
       salesman_id
FROM customers
WHERE grade < 300
ORDER BY customer_id ASC;


SELECT c.customer_id,
       c.customer_name
FROM customers c
JOIN salesman s
ON c.salesman_id = s.salesman_id
WHERE s.commission > 0.12;


SELECT a.order_no,
       a.order_date,
       a.purchase_amount,
       b.customer_name,
       c.salesman_name,
       c.commission
FROM orders a
INNER JOIN customers b
ON a.customer_id = b.customer_id
INNER JOIN salesman c
ON a.salesman_id = c.salesman_id;
